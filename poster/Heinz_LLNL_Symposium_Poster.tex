    \LARGE
    \begin{block}{Traditional vs Covariant Treatment of Errors}
      \Large
      \begin{itemize}
        \item Current errors given bin by bin
        \item Statistical errors assumed to be uncorrelated
        \item Other uncorrelated systematics lumped in with statistical errors
        \item Any correlated systematics are fully correlated across all bins
        \item Covariance matrix defined as:
      \end{itemize}
      \begin{center}
        $C_{ij} = \rho_{ij} \sigma_i \sigma_j$
      \end{center}
      \begin{itemize}
        \item $\rho_{ij}$ is the covariance coefficient between the $i^{th}$ and $j^{th}$ bins
        \item For uncorrelated errors, $\rho_{ij} = \delta_{ij}$
        \item For fully correlated errors, $\rho_{ij} = 1$
        \item Covariance matrix formulation of errors allows for non-uniform covariance
        \item In particular, it supports errors with a correlation length
      \end{itemize}
    \end{block}

    \vspace{2.0cm}

    \LARGE
    \begin{block}{Apply Covariance Formalism to Previous Result}
      \Large
      \begin{itemize}
        \item ALICE measurement of $v_2$ provides good test for covariance formalism equivalence
        \item Has uncorrelated, correlated, and shape uncertainties
        \item $p$-values published in paper are obtained by minimizing $\chi^2$
        \item Equivalent calculation of minimized $\chi^2$ via covariance matrices looks like:
      \end{itemize}
      \begin{center}
        $\chi^2_{min} = \Delta^T C^{-1} \Delta$
      \end{center}
      \begin{itemize}
        \item Resulting $p$-values are exactly equal:
      \end{itemize}
      \begin{table}[h]
      \begin{center}
      \begin{tabular}{ l r r }
        % & \multicolumn{2}{c}{$p$-value} \\
      $p_T$ range & $\chi^2$ minimization & Covariance matrix \\
      \hline
      30-100 GeV & 0.1247858497 & 0.1247858497 \\ 
      30-60 GeV & 0.0685915881 & 0.0685915881 \\ 
      60-100 GeV & 0.0211009165 & 0.0211009165 \\
      % \hline
      \end{tabular}
      \caption{$p$-values reported to machine level significance calculated using traditional $\chi^2$ minimization and covariance method.}
      \end{center}
      \end{table}
    \end{block}

    \LARGE
    \begin{block}{Incorperating Covariance into HEPData}
      \Large
      \begin{itemize}
        \item Current HEPData schema lacks support for full covariance matrices
        \item Adapted schema to support:
        \begin{itemize}
          \Large
          \item (A)symmetric uncorrelated errors
          \item (A)symmetric fully correlated errors
          \item Full covariance matrices
        \end{itemize}
        \item Distinct treatment of uncorrelated and fully correlated errors to save space
        \item Would like to see adoption of this modified schema by the HEPData community for future experimental results
      \end{itemize}
    \end{block}
